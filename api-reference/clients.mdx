# API de Clientes

Gerencie os clientes da sua empresa através destes endpoints.

## Listar Clientes

<ParamField path="GET" endpoint="/api/v1/clients">
  Lista todos os clientes da empresa
</ParamField>

### Parâmetros de Query

<ParamField query="page" type="number" default="1">
  Número da página para paginação
</ParamField>

<ParamField query="limit" type="number" default="20">
  Número de registros por página (máximo: 100)
</ParamField>

<ParamField query="search" type="string">
  Termo de busca (pesquisa em nome, código, cidade e email)
</ParamField>

### Resposta

<ResponseField name="data" type="array">
  Lista de clientes
</ResponseField>

<ResponseField name="page" type="number">
  Página atual
</ResponseField>

<ResponseField name="limit" type="number">
  Registros por página
</ResponseField>

<ResponseField name="total" type="number">
  Total de registros
</ResponseField>

## Criar Cliente

<ParamField path="POST" endpoint="/api/v1/clients">
  Cria um novo cliente
</ParamField>

### Body da Requisição

<ParamField body="nome" type="string" required>
  Nome do cliente
</ParamField>

<ParamField body="codigo" type="string">
  Código único do cliente
</ParamField>

<ParamField body="email" type="string">
  Email do cliente (deve ser válido)
</ParamField>

<ParamField body="telefone" type="string">
  Telefone do cliente
</ParamField>

<ParamField body="cidade" type="string">
  Cidade do cliente
</ParamField>

<ParamField body="estado" type="string">
  Estado (UF) do cliente
</ParamField>

<ParamField body="cnpj_cpf" type="string">
  CNPJ ou CPF do cliente
</ParamField>

<ParamField body="endereco" type="string">
  Endereço completo
</ParamField>

<ParamField body="bairro" type="string">
  Bairro
</ParamField>

<ParamField body="observacao" type="string">
  Observações adicionais
</ParamField>

<ParamField body="ativo" type="boolean" default="true">
  Status do cliente
</ParamField>

<RequestExample>
```json
{
  "nome": "Nova Empresa Ltda",
  "codigo": "CLI002",
  "email": "contato@novaempresa.com.br",
  "telefone": "(11) 91234-5678",
  "cidade": "São Paulo",
  "estado": "SP",
  "cnpj_cpf": "98.765.432/0001-10",
  "endereco": "Av. Paulista, 1000",
  "bairro": "Bela Vista",
  "observacao": "Potencial cliente grande",
  "ativo": true
}
```
</RequestExample>
